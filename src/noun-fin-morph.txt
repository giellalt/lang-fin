! Basic paradigms
! ===============

! Basic vowel stems
! -----------------

LEXICON strong-v-stem
 nom-strong ;
 gae-strong ;
 +Sg:%>i2 com ;
 +Pl:%>i2 com ;

LEXICON weak-v-stem
 +Sg+Gen:%>n K ;
 +Sg+Acc:%>n K ;
 +Pl+Nom:%>t K ;
 +Pl+Acc:%>t K ;
 +Sg: sgpl-cases ;     ! these two lexica for cases with
 +Pl:%>i2 sgpl-cases ; ! the same affix in sg and pl
 +Pl+Ins:%>i2%>n K ;   ! Marginal case, in use only for some words.
                       ! Todo: identify these, and remove Ins from the
                       ! general weak-v-stem lexicon.


! Sublexica for the vowel stems
! - - - - - - - - - - - - - - - 

LEXICON nom-strong
 +Sg+Nom: K ;

LEXICON gae-strong
 +Sg+Nom: Px ;  ! mattoni
 +Sg+Gen: Px ;  ! mattoni
 +Sg+Acc: Px ;  ! mattoni
 +Pl+Nom: Px ;  ! mattoni
 +Pl+Acc: Px ;  ! mattoni
 +Sg+Ess:%>n%^A PxxK ;
 +Pl+Ess:%>i2%>n%^A PxxK ;
 
LEXICON sgpl-cases
 +Ine:%>ss%^A PxxK ;
 +Ela:%>st%^A PxxK ;
 +Ade:%>ll%^A PxxK ;
 +All:%>lle PxxK ;
 +Abl:%>lt%^A PxxK ;
 +Tra:%>ksi K ;
 +Tra:%>kse Pxx ;
 +Abe:%>tt%^A PxxK ;

LEXICON com
 +Com:%>ne Pxx ; ! Not to K, since Px must be present
                 ! Note that adjectives have -ine without Px


! Lexica for the non-uniform cases
! --------------------------------

LEXICON short-par
 +Sg+Par:%>%^A PxxK ;
! does not handle valoaan vs. *salaaan

LEXICON long-par
 +Sg+Par:%>t%^A PxxK ;

LEXICON short-ill
 +Sg+Ill:%>%^V n-PxK ;

LEXICON long-ill
 +Sg+Ill:%>h%^V n-PxK ;

LEXICON longer-ill
 +Sg+Ill:%>s%^V%^V n-PxK ;

LEXICON oldgenpl
 +Gen:%> n-PxK ;

LEXICON shortgenpl
 +Gen:%>e n-PxK ;

LEXICON longgenpl
 d-longgenpl ;
 tt-longgenpl ; 

LEXICON d-longgenpl
 +Gen:%>de n-PxK ;

LEXICON tt-longgenpl
 +Gen:%>tte n-PxK ;

LEXICON consgenpl
 +Gen:%>te n-PxK ;

LEXICON shortparpl
 +Par:%>%^A PxxK ;

LEXICON longparpl
 +Par:%>%t^A PxxK ;

LEXICON shortillpl
 +Ill:%>hi n-PxK ;

LEXICON shorterillpl
 +Ill:%>i2 n-PxK ;		

LEXICON longerillpl
 +Pl+Ill:%>s%^V%^V n-PxK ;

! Sublexica for possessive suffixes
! ---------------------------------

LEXICON n-PxK
 :n K ;
 Px ;

LEXICON PxK ! only -nsA
 K ;
 Px ;

LEXICON PxxK ! also -Vn
 K ;
 Px ;
 Px-Vn ;

LEXICON Pxx
 Px ;
 Px-Vn ;

LEXICON Px
 +PxSg1:%>ni K ;
 +PxSg2:%>si K ;
 +PxSg3:%>ns%^A K ;
 +PxPl1:%>mme K ;
 +PxPl2:%>nne K ;
 +PxPl3:%>ns%^A K ;


LEXICON Px-Vn
 +PxSg3:%>%^Vn K ;
 +PxPl3:%>%^Vn K ;
